130  public void Process(IList<IAsset> Assets, BundleResponse Response)
131  {
132  Contract.Requires<ArgumentNullException>(Response != null, "Response");
133  if (Assets == null || Assets.Count == 0)
134  return;
135  foreach (IFilter Filter in Filters.Where(x => x.TimeToRun == RunTime.PreTranslate))
136  {
137  Assets = Filter.Filter(Assets);
138  }
139  foreach (ITranslator Translator in Translators)
140  {
141  Assets = Translator.Translate(Assets);
142  }
143  foreach (RunTime Item in RunOrder)
144  {
145  foreach (IFilter Filter in Filters.Where(x => x.TimeToRun == Item))
146  {
147  Assets = Filter.Filter(Assets);
148  }
149  }
150  string Content = Assets.OrderBy(x => x.Path).ToString(x => x.ToString(), "");
151  foreach (IContentFilter Filter in ContentFilters)
152  {
153  Content = Filter.Filter(Content);
154  }
155  System.Collections.Generic.List<BundleFile> Files = new System.Collections.Generic.List<BundleFile>();
156  foreach (IAsset Asset in Assets)
157  {
158  if (Asset.Path.StartsWith("~") || Asset.Path.StartsWith("/"))
159  {
160  Files.Add(new BundleFile(Asset.Path, new VirtualFileHack(Asset.Path)));
161  Files.Add(Asset.Included.Where(x => x.Path.StartsWith("~") || x.Path.StartsWith("/")).Select(x => new BundleFile(x.Path, new VirtualFileHack(x.Path))));
162  }
163  }
164  Response.Content = Content;
165  Response.Files = Files.OrderBy(x => x.VirtualFile.VirtualPath);
166  Response.ContentType = Assets.First().Type == AssetType.CSS ? "text/css" : "text/javascript";
167  Response.Cacheability = HttpCacheability.ServerAndPrivate;
168  }

187  private void CreateBundles(DirectoryInfo Directory)
188  {
189  if (Directory == null || !Directory.Exists || string.IsNullOrEmpty(Directory.FullName))
190  return;
191  string BundleDirectory = Directory.FullName.Replace(new DirectoryInfo("~/").FullName, "~/").Replace("\\", "/");
192  StyleBundle Bundle = new StyleBundle(BundleDirectory + "/bundle/css");
193  Bundle.Transforms.Clear();
194  Bundle.Transforms.Add(new Transformer(this));
195  if (Directory.Exists)
196  {
197  foreach (string Value in FileTypes[AssetType.CSS])
198  {
199  Bundle.IncludeDirectory(BundleDirectory, "*." + Value, true);
200  }
201  }
202  ScriptBundle Bundle2 = new ScriptBundle(BundleDirectory + "/bundle/js");
203  Bundle2.Transforms.Clear();
204  Bundle2.Transforms.Add(new Transformer(this));
205  if (Directory.Exists)

207  foreach (string Value in FileTypes[AssetType.Javascript])
208  {
209  Bundle2.IncludeDirectory(BundleDirectory, "*." + Value, true);
210  }
211  }
212  BundleTable.Bundles.Add(Bundle);
213  BundleTable.Bundles.Add(Bundle2);
214  foreach (DirectoryInfo SubDirectory in Directory.EnumerateDirectories("*", System.IO.SearchOption.TopDirectoryOnly))
215  {
216  CreateBundles(SubDirectory);
