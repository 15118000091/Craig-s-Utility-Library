99  public string SetupEndMethod(MethodInfo Method, Type BaseType, string ReturnValueName)
100  {
101  StringBuilder Builder = new StringBuilder();
102  if (Mapper[BaseType] != null && Method.Name.StartsWith("get_", StringComparison.Ordinal))
103  {
104  foreach (IMapping Mapping in Mapper[BaseType])
105  {
106  IProperty Property = Mapping.Properties.FirstOrDefault(x => x.Name == Method.Name.Replace("get_", ""));
107  if (Property != null)
108  {
109  if (Property is IManyToOne || Property is IMap)
110  Builder.AppendLine(SetupSingleProperty(ReturnValueName, Property));
111  else if (Property is IIEnumerableManyToOne || Property is IManyToMany)
112  Builder.AppendLine(SetupIEnumerableProperty(ReturnValueName, Property));
113  else if (Property is IListManyToMany || Property is IListManyToOne)
114  Builder.AppendLine(SetupListProperty(ReturnValueName, Property));
115  return Builder.ToString();
116  }
117  }
118  }
119  return Builder.ToString();
120  }

232  private string SetupFields(Type Type)
233  {
234  Fields = new List<IProperty>();
235  StringBuilder Builder = new StringBuilder();
236  if (Mapper[Type] != null)
237  {
238  foreach (IMapping Mapping in Mapper[Type])
239  {
240  foreach (IProperty Property in Mapping.Properties)
241  {
242  if (Property is IManyToOne || Property is IMap)
243  {
244  if (Fields.FirstOrDefault(x => x.DerivedFieldName == Property.DerivedFieldName) == null)
245  {
246  Fields.Add(Property);
247  Builder.AppendLineFormat("private {0} {1};", Property.Type.GetName(), Property.DerivedFieldName);
248  Builder.AppendLineFormat("private bool {0};", Property.DerivedFieldName + "Loaded");
249  }
250  }
251  else if (Property is IIEnumerableManyToOne || Property is IManyToMany)
252  {
253  if (Fields.FirstOrDefault(x => x.DerivedFieldName == Property.DerivedFieldName) == null)
254  {
255  Fields.Add(Property);
256  Builder.AppendLineFormat("private {0} {1};", typeof(IEnumerable<>).MakeGenericType(Property.Type).GetName(), Property.DerivedFieldName);
257  Builder.AppendLineFormat("private bool {0};", Property.DerivedFieldName + "Loaded");
258  }
259  }
260  else if (Property is IListManyToOne || Property is IListManyToMany)
261  {
262  if (Fields.FirstOrDefault(x => x.DerivedFieldName == Property.DerivedFieldName) == null)
263  {
264  Fields.Add(Property);
265  Builder.AppendLineFormat("private {0} {1};", typeof(List<>).MakeGenericType(Property.Type).GetName(), Property.DerivedFieldName);
266  Builder.AppendLineFormat("private bool {0};", Property.DerivedFieldName + "Loaded");
267  }
268  }
269  }
270  }
271  }
