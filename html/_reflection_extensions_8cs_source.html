417  {
418  if (Object == null)
419  return default(T);
420  Type ObjectType = Object.GetType();
421  T ClassInstance = ObjectType.Create<T>();
422  foreach (PropertyInfo Property in ObjectType.GetProperties())
423  {
424  try
425  {
426  if (Property.CanRead

428  && SimpleTypesOnly
429  && Property.PropertyType.IsValueType)

431  else if (!SimpleTypesOnly

434  Property.SetValue(ClassInstance, Property.GetValue(Object, null), null);

436  catch { }

440  {

444  Field.SetValue(ClassInstance, Field.GetValue(Object));

682  {
683  Contract.Requires<ArgumentNullException>(Property != null, "Property");
684  string PropertyName = Property.PropertyName();
685  string[] SplitName = PropertyName.Split(new string[] { "." }, StringSplitOptions.RemoveEmptyEntries);
686  if (SplitName.Length == 0)
687  return null;
688  PropertyInfo PropertyInfo = typeof(ClassType).GetProperty(SplitName[0]);
689  ParameterExpression ObjectInstance = Expression.Parameter(PropertyInfo.DeclaringType, "x");
690  ParameterExpression PropertySet = Expression.Parameter(typeof(DataType), "y");
691  ConstantExpression DefaultConstant = Expression.Constant(((object)null).To(PropertyInfo.PropertyType, null), PropertyInfo.PropertyType);
692  MethodCallExpression SetterCall = null;
693  MemberExpression PropertyGet = null;
694  if (SplitName.Length > 1)
695  {
696  PropertyGet = Expression.Property(ObjectInstance, PropertyInfo);
697  for (int x = 1; x < SplitName.Length - 1; ++x)
698  {
699  PropertyInfo = PropertyInfo.PropertyType.GetProperty(SplitName[x]);
700  if (PropertyInfo == null)
701  throw new NullReferenceException("PropertyInfo can't be null");
702  PropertyGet = Expression.Property(PropertyGet, PropertyInfo);
703  }
704  PropertyInfo = PropertyInfo.PropertyType.GetProperty(SplitName[SplitName.Length - 1]);
705  }
706  MethodInfo SetMethod = PropertyInfo.GetSetMethod();
707  if (SetMethod != null)
708  {
709  if (PropertyInfo.PropertyType != typeof(DataType))
710  {
711  MethodInfo ConversionMethod = typeof(TypeConversionExtensions).GetMethods().FirstOrDefault(x => x.ContainsGenericParameters
712  && x.GetGenericArguments().Length == 2
713  && x.Name == "To"
714  && x.GetParameters().Length == 2);
715  ConversionMethod = ConversionMethod.MakeGenericMethod(typeof(DataType), PropertyInfo.PropertyType);
716  MethodCallExpression Convert = Expression.Call(ConversionMethod, PropertySet, DefaultConstant);
717  if (PropertyGet == null)
718  SetterCall = Expression.Call(ObjectInstance, SetMethod, Convert);
719  else
720  SetterCall = Expression.Call(PropertyGet, SetMethod, Convert);
721  return Expression.Lambda<Action<ClassType, DataType>>(SetterCall, ObjectInstance, PropertySet);
722  }
723  if (PropertyGet == null)
724  SetterCall = Expression.Call(ObjectInstance, SetMethod, PropertySet);
725  else
726  SetterCall = Expression.Call(PropertyGet, SetMethod, PropertySet);
727  }
