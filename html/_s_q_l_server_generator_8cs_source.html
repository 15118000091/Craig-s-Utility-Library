406  public IBatch Save<PrimaryKeyType>(T Object)
407  {
408  IBatch TempBatch = QueryProvider.Batch(Source);
409  IProperty<T> IDProperty = ((IProperty<T>)Mapping.IDProperties.FirstOrDefault());
410  PrimaryKeyType IDValue = IDProperty.GetValue(Object).To(default(PrimaryKeyType));
411 
412  GenericEqualityComparer<PrimaryKeyType> Comparer = new GenericEqualityComparer<PrimaryKeyType>();
413  IParameter Param1 = null;
414  if (Comparer.Equals(IDValue, default(PrimaryKeyType)))
415  {
416  return TempBatch.AddCommand(Insert(Object));
417  }
418  if (IDProperty.AutoIncrement)
419  {
420  return TempBatch.AddCommand(Update(Object));
421  }
422  if (typeof(PrimaryKeyType).Is(typeof(string)))
423  Param1 = new StringEqualParameter(IDValue.ToString(), IDProperty.FieldName, IDValue.ToString().Length, IDProperty.FieldName, Source.ParameterPrefix);
424  else
425  Param1 = new EqualParameter<PrimaryKeyType>(IDValue, IDProperty.FieldName, IDProperty.FieldName, Source.ParameterPrefix);
426  if (Any(Param1).Execute()[0].Count == 0)
427  {
428  return TempBatch.AddCommand(Insert(Object));
429  }
430  return TempBatch.AddCommand(Update(Object));
431  }
